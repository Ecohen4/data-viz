lgas<-read.csv("lgas.csv")
health<-read.csv("Health_Facility.csv")
mgdata=merge(lgas,health,by="unique_lga")
south<-subset(mgdata,zone=="Southwest"|zone=="Southeast"|zone=="South-South")

tot.nurse=aggregate(num_nurses_fulltime~state, data=south, sum)
tot.doctor=aggregate(num_doctors_fulltime~state, data=south, sum)
statearea=aggregate(area_sq_km~state, data=lgas, sum)
statepop=aggregate(pop_2006~state, data=lgas, sum)

result=merge(tot.nurse,tot.doctor, by="state")
result=merge(result,statearea,by="state")
result=merge(result,statepop,by="state")
result["nurses_per_area"]<-NA
result["nurses_per_capita"]<-NA
result["doctors_per_area"]<-NA
result["doctors_per_capita"]<-NA

result$nurses_per_area=result$num_nurses_fulltime/result$area_sq_km
result$nurses_per_capita=result$num_nurses_fulltime/result$pop_2006
result$doctors_per_area=result$num_doctors_fulltime/result$area_sq_km
result$doctors_per_capita=result$num_doctors_fulltime/result$pop_2006

result=result[order(-result$pop_2006), ]
attach(result)
