library(ggplot2)
data=read.xls('w4063 hw2 data.xlsx')
data=data[data$RO<=17,]

# base plot
myplot=ggplot(data,aes(x=RO,y=case.fatality.rate,color=primary.mode.of.transmission))+geom_point(shape=19,size=4.5)+xlab('average basic reproduction number (R0)')+ylab('case fatality rate')

# grid
myplot=myplot+theme_bw()

#title
myplot=myplot+ggtitle('The Microbe-scope')+theme(plot.title=element_text(hjust=0))

# axis
myplot=myplot+scale_y_continuous(breaks=c(0,0.001,0.01,seq(0,1,0.1)))
myplot=myplot+scale_x_continuous(breaks=c(seq(0,17,1)))

# label for each data point
myplot=myplot+geom_text(aes(label=primary.mode.of.transmission),size=2.5,hjust=-0.35)


# legend
myplot=myplot+theme(legend.position='top')

# annotate
myplot=myplot+annotate('rect',xmin=0,xmax=17,ymin=0,ymax=0.01,alpha=0.1)

myplot=myplot+annotate('rect',xmin=0,xmax=17,ymin=0.2,ymax=0.5,alpha=0.1)

myplot=myplot+annotate('rect',xmin=0.999,xmax=1.001,ymin=0,ymax=1.1,color='black',alpha=0)+annotate('rect',xmin=4.999,xmax=5.001,ymin=0,ymax=1.1,color='grey')+annotate('rect',xmin=6.999,xmax=7.001,ymin=0,ymax=1.1,color='grey')+annotate('rect',xmin=11.999,xmax=12.001,ymin=0,ymax=1.1,color='grey')+annotate('rect',xmin=0,xmax=17,ymin=0.00999,ymax=0.01001,color='black',alpha=0)

myplot=myplot+annotate('text',x=0.5,y=1.05,label='not very',size=3,color='grey')+annotate('text',x=3,y=1.05,label='quite contagious',size=3,color='grey')+annotate('text',x=6,y=1.05,label='very',size=3,color='grey')+annotate('text',x=9.5,y=1.05,label='highly',size=3,color='grey')+annotate('text',x=14.5,y=1.05,label='vaccinate now!',size=3,color='grey')+annotate('text',x=18,y=0.001,label='not too deadly\nhigh-risk groups\n(infants, the aged)',size=2.5,color='grey')+annotate('text',x=18,y=0.1,label='quite deadly\nunlucky or unhealthy',size=2.5,color='grey')+annotate('text',x=18,y=0.35,label='deadly\nhigh chance',size=2.5,color='grey')+annotate('text',x=18,y=0.8,label='extremely deadly\ndeath likely',size=2.5,color='grey')
myplot=myplot+scale_color_discrete(name="PRIMARY TRANSMISSION METHOD")
