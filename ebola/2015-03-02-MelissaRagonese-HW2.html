Ensure that data entries for fatality rate and contagiousness are of numeric class. Combine the airborne and airborne droplet entries to have one airborne category.

#eliminate non-numeric entries
germsnumeric <- germs[3:41,]

#function to remove percents from entries
dropPct<-function(df, var){
  dropPct<-strsplit(as.character(df[[var]]), split="%")
  df[[var]]<-as.numeric(dropPct)
  return(df[[var]])
  }

#remove percents from fatality rates
germsnumeric[["case.fatality.rate"]]<-dropPct(germsnumeric, "case.fatality.rate")

#convert factor entries for fatality and contagiousness to numeric
germinfect <- germsnumeric$average.basic.reproductive.rate
germinfect <- as.numeric(levels(germinfect))[germinfect]
## Warning: NAs introduced by coercion
germsnumeric$average.basic.reproductive.rate <- germinfect

#replace 'airborne droplet' with 'airborne' to create one airborne category
germsnumeric$primary.mode.of.transmission <- revalue(germsnumeric$primary.mode.of.transmission, c("airborne droplet" = "airborne"))

The plot (sorry)

#plot: deadlieness on x axis, contagiousness on y axis, color by transmission, continuous x scale 0-17, continuous y scale 0-100 with breaks at 0, 0.1, and 10.
ggplot(germsnumeric, aes(x=average.basic.reproductive.rate, y=case.fatality.rate, colour=primary.mode.of.transmission)) + geom_point(size=3) + geom_text(aes(label=X), size=3) + scale_y_continuous(breaks=c(0,0.1,10,seq(20,100,by=10))) + scale_x_continuous(limits=c(-3,17)) + theme(axis.ticks.length=unit(1,"cm")) + labs(title = "MicrobeScope Recreation!") + ylab("DEADLINESS (Percent of infected people who died)") + xlab("CONTAGIOUSNESS (Average Basic Reproductive Rate)") + scale_colour_discrete(name="Primary Mode of Transmission")
