Loading Data and Libraries

For our initial analysis on the Delhi Data, we will be using just the plyr and ggplot2 libraries. This markdown shows the steps taken to manipulate the data into a more manageable format, and provides plots that form the basis of our initial inferences.

library(plyr)
library(ggplot2)

setwd("C:/Users/suresh/Documents/Github/Data-Viz-Project")
Delhi_Demand_Hourly <- read.csv("Delhi_Discoms_Hourly_Demand.csv")
Delhi_Weather_Hourly <- read.csv("Delhi_Discom_Hourly.csv")

#Subsetting data by year
Delhi_2012_Demand <- subset(Delhi_Demand_Hourly,YR==2012,select=city:MW)
Delhi_2012_Weather <- subset(Delhi_Weather_Hourly,YR==2012,select=hours:DEW.POINT)

#Adding Date variable
Delhi_2012_Demand$Date <- as.character(paste(Delhi_2012_Demand$YR,Delhi_2012_Demand$M,Delhi_2012_Demand$D,sep="-"))
Delhi_2012_Weather <- rename(Delhi_2012_Weather,c("hours"="Date"))

#Turning Dates into POSIXct object
Delhi_2012_Demand$Date <- as.POSIXct(Delhi_2012_Demand$Date)
Delhi_2012_Weather$Date <- as.POSIXct(Delhi_2012_Weather$Date)

#Ordering the Demand Data by Date instead of by hour
Delhi_2012_Demand <- Delhi_2012_Demand[order(as.POSIXlt(Delhi_2012_Demand$Date)),]

Now that the data has been organized by Date and has a valid timestamp element, we can begin to break it down into more manageable sections.

#Function to put multiple plots on a single graph
multiplot <- function(..., plotlist=NULL, file, cols=1, layout=NULL) {
  library(grid)
  
  # Make a list from the ... arguments and plotlist
  plots <- c(list(...), plotlist)
  
  numPlots = length(plots)
  
  # If layout is NULL, then use 'cols' to determine layout
  if (is.null(layout)) {
    # Make the panel
    # ncol: Number of columns of plots
    # nrow: Number of rows needed, calculated from # of cols
    layout <- matrix(seq(1, cols * ceiling(numPlots/cols)),
                     ncol = cols, nrow = ceiling(numPlots/cols))
  }
  
  if (numPlots==1) {
    print(plots[[1]])
    
  } else {
    # Set up the page
    grid.newpage()
    pushViewport(viewport(layout = grid.layout(nrow(layout), ncol(layout))))
    
    # Make each plot, in the correct location
    for (i in 1:numPlots) {
      # Get the i,j matrix positions of the regions that contain this subplot
      matchidx <- as.data.frame(which(layout == i, arr.ind = TRUE))
      
      print(plots[[i]], vp = viewport(layout.pos.row = matchidx$row,
                                      layout.pos.col = matchidx$col))
    }
  }
}
