#---- make data prepared ----
setwd("~/Documents/R_path")
NMIS<-read.csv("Health_Mopup_and_Baseline_NMIS_Facility.csv")
LGA<-read.csv("lgas.csv")
new<-merge(NMIS,LGA,by="unique_lga",sort=F)

#---- select tags that belong to South zones ----
tag=vector()
for (i in 1:dim(new)[1]){
    if(new$zone[i]=="South-South" || new$zone[i]=="Southwest" || new$zone[i]=="Southeast")
        tag[i]=TRUE
    else
        tag[i]=FALSE
}  
new_data=new[tag,]

#---- calculate doctors and nurses ----
new_data2=subset(new_data,!is.na(num_doctors_fulltime))
agg_doctor=aggregate(new_data2$num_doctors_fulltime,as.data.frame(new_data2$state),sum)
names(agg_doctor)=c("state","sum.of.doctor")
new_data3=subset(new_data,!is.na(num_nurses_fulltime))
agg_nurse=aggregate(new_data3$num_nurses_fulltime,as.data.frame(new_data3$state),sum)
names(agg_nurse)=c("state","sum.of.nurse")
agg1=merge(agg_doctor,agg_nurse,by="state")

#---- calculate population and area ----
agg2=unique(data.frame(LGA$state,LGA$pop_2006,LGA$area_sq_km))
names(agg2)=c("state","population","area")
agg2=aggregate(cbind(agg2$population,agg2$area),as.data.frame(agg2$state),sum)
names(agg2)=c("state","population","area")
agg=merge(agg1,agg2,by="state")

#---- calculate capita and sort descending by population ----
per_capita<-(agg$sum.of.doctor+agg$sum.of.nurse)/agg$population
per_area_value<-(agg$sum.of.doctor+agg$sum.of.nurse)/agg$area
agg=data.frame(agg,per_capita,per_area_value)
agg=agg[order(agg$population,decreasing=T),]
agg<-data.frame(agg)
show(agg)
